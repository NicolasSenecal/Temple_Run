66  { }

68  bool operator < (const ColladaMeshIndex& p) const
69  {
70  if( mMeshID == p.mMeshID)
71  {
72  if( mSubMesh == p.mSubMesh)
73  return mMaterial < p.mMaterial;
74  else
75  return mSubMesh < p.mSubMesh;

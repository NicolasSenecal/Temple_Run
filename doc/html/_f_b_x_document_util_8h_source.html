78 template <typename T>
79 inline const T* ProcessSimpleConnection(const Connection& con,
80  bool is_object_property_conn,
81  const char* name,
82  const Element& element,
83  const char** propNameOut = NULL)
84 {
85  if (is_object_property_conn && !con.PropertyName().length()) {
86  DOMWarning("expected incoming " + std::string(name) +
87  " link to be an object-object connection, ignoring",
88  &element
89  );
90  return NULL;
91  }
92  else if (!is_object_property_conn && con.PropertyName().length()) {
93  DOMWarning("expected incoming " + std::string(name) +
94  " link to be an object-property connection, ignoring",
95  &element
96  );
97  return NULL;
98  }
99 
100  if(is_object_property_conn && propNameOut) {
101  // note: this is ok, the return value of PropertyValue() is guaranteed to
102  // remain valid and unchanged as long as the document exists.
103  *propNameOut = con.PropertyName().c_str();
104  }
105 
106  const Object* const ob = con.SourceObject();
107  if(!ob) {
108  DOMWarning("failed to read source object for incoming" + std::string(name) +
109  " link, ignoring",
110  &element);
111  return NULL;
112  }
113 
114  return dynamic_cast<const T*>(ob);
115 }
